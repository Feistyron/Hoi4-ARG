on_actions = {

	# Event 1 — Захват столицы ARG
	on_capital_captured = {
		effect = {
			if = {
				limit = { PREV = { tag = ARG } }
				PREV = { country_event = { id = arg_special.1 } }
			}
		}
	}

	# Event 2, 4 — Капитуляция Уругвая и Венесуэлы
	on_capitulation = {
		effect = {
			# Уругвай капитулировал
			if = {
				limit = { tag = URY }
				ARG = { country_event = { id = arg_special.2 } }
			}
			# Венесуэла капитулировала
			if = {
				limit = { tag = VEN }
				ARG = { country_event = { id = arg_special.4 } }
			}
		}
	}

	# Event 3 — Аннексия Парагвая
	on_annex = {
		effect = {
			if = {
				limit = {
					FROM = { tag = PRY }
					ROOT = { tag = ARG }
				}
				ROOT = { country_event = { id = arg_special.3 } }
			}
		}
	}

}
